1"""Safe loading of on-disk FAISS indexes. 

2 

3LangChain's ``FAISS.load_local(..., allow_dangerous_deserialization=True)`` 

4deserializes the ``<index>.pkl`` docstore companion with a raw 

5``pickle.load`` (see ``langchain_community/vectorstores/faiss.py``). Pickle 

6executes arbitrary code during deserialization, so a tampered ``.pkl`` 

7dropped into the index directory yields arbitrary code execution when the 

8index is next loaded. 

9 

10The ``.faiss`` and ``.pkl`` files are loaded independently with no 

11cross-check, so a content checksum over only the ``.faiss`` file (which is 

12what the file-integrity system records) provides **no** protection against a 

13``.pkl``-only swap. Rather than gate the dangerous load behind a checksum, 

14this module removes the dangerous deserialization entirely: it replicates 

15``load_local`` but unpickles the docstore through a restricted unpickler that 

16only permits the two classes a legitimate FAISS docstore contains 

17(:class:`InMemoryDocstore` and :class:`Document`). Any other global 

18(``os.system``, ``subprocess.Popen``, a crafted ``__reduce__`` payload, …) 

19raises :class:`pickle.UnpicklingError` before it can execute. 

20 

21The payload is the 2-tuple ``(docstore, index_to_docstore_id)`` written by 

22``FAISS.save_local``; the metadata it carries is plain JSON-ish scalars 

23(str/int/float/bool/None/list/dict), none of which require a class global to 

24unpickle. If a future LangChain/Pydantic version changes the on-disk format 

25to need additional classes, loading fails closed (a clear error) rather than 

26silently — and the round-trip test in 

27``tests/research_library/services/test_faiss_safe_load.py`` will catch it so 

28the allow-list can be widened deliberately. 

29""" 

30 

31import pickle 

32from pathlib import Path 

33from typing import Any 

34 

35from langchain_community.vectorstores import FAISS 

36from langchain_community.vectorstores.faiss import dependable_faiss_import 

37from langchain_core.embeddings import Embeddings 

38 

39# (module, qualname) pairs that a legitimate FAISS docstore pickle resolves 

40# via ``Unpickler.find_class``. Determined empirically by round-tripping a 

41# real ``save_local`` payload with varied metadata; only these two appear. 

42_ALLOWED_GLOBALS = frozenset( 

43 { 

44 ("langchain_community.docstore.in_memory", "InMemoryDocstore"), 

45 ("langchain_core.documents.base", "Document"), 

46 } 

47) 

48 

49 

50# Subclass the PURE-PYTHON unpickler (``pickle._Unpickler``), NOT the default 

51# C ``pickle.Unpickler``. We need to override ``get_extension`` to refuse the 

52# copyreg extension opcodes (EXT1/EXT2/EXT4), and only the pure-Python unpickler 

53# routes those opcodes through an overridable ``get_extension`` method — the C 

54# unpickler resolves them internally. (Why ``get_extension`` and not just 

55# ``find_class``: both unpicklers normally route EXT through ``find_class`` on a 

56# cold cache, but ``get_extension`` short-circuits on a warm process-global 

57# ``copyreg._extension_cache`` and skips ``find_class`` entirely. Refusing 

58# ``get_extension`` outright closes that path regardless of cache/registry 

59# state, instead of relying on the extension registry happening to be empty.) 

60# ``_Unpickler`` has been the stable pure-Python name since Python 3.0; if a 

61# future runtime drops it, this fails loudly at import (caught by tests) rather 

62# than silently falling back to the C unpickler. 

63class _RestrictedFaissUnpickler(pickle._Unpickler): 

64 """Unpickler that only resolves the classes a FAISS docstore contains. 

65 

66 Defense is by construction: a malicious pickle can never obtain an 

67 attacker-chosen callable to invoke, because every opcode that resolves a 

68 global or callable is refused or allow-listed: 

69 

70 - ``find_class`` (GLOBAL/STACK_GLOBAL/INST/OBJ, then REDUCE/NEWOBJ/BUILD) 

71 allow-lists only ``InMemoryDocstore`` and ``Document`` — both inert. 

72 - ``get_extension`` (copyreg EXT1/EXT2/EXT4) is refused outright; a 

73 legitimate FAISS docstore pickle never uses copyreg extensions. 

74 - ``persistent_load`` (PERSID/BINPERSID) is refused outright; ``save_local`` 

75 never emits persistent ids. 

76 """ 

77 

78 def find_class(self, module: str, name: str) -> Any: 

79 if (module, name) in _ALLOWED_GLOBALS: 

80 return super().find_class(module, name) 

81 raise pickle.UnpicklingError( 

82 f"Refusing to unpickle disallowed object '{module}.{name}' from " 

83 f"FAISS docstore (possible tampering)." 

84 ) 

85 

86 def get_extension(self, code: int) -> Any: 

87 raise pickle.UnpicklingError( 

88 f"Refusing to resolve copyreg extension code {code} from FAISS " 

89 f"docstore (possible tampering)." 

90 ) 

91 

92 def persistent_load(self, pid: Any) -> Any: 

93 raise pickle.UnpicklingError( 

94 "Refusing to resolve a persistent id from FAISS docstore " 

95 "(possible tampering)." 

96 ) 

97 

98 

99def safe_load_faiss( 

100 folder_path: str, 

101 embeddings: Embeddings, 

102 *, 

103 index_name: str, 

104 normalize_L2: bool = False, 

105) -> FAISS: 

106 """Load a FAISS index from disk without dangerous deserialization. 

107 

108 Drop-in replacement for ``FAISS.load_local(..., 

109 allow_dangerous_deserialization=True)`` for the index/docstore layout 

110 this project writes via ``FAISS.save_local``. 

111 

112 Args: 

113 folder_path: Directory containing ``<index_name>.faiss`` and 

114 ``<index_name>.pkl``. 

115 embeddings: Embeddings used for subsequent queries. 

116 index_name: Base name of the index files (the ``.faiss`` stem). 

117 normalize_L2: Whether vectors are L2-normalized (must match how the 

118 index was built). 

119 

120 Returns: 

121 A :class:`FAISS` vector store. 

122 

123 Raises: 

124 pickle.UnpicklingError: If the ``.pkl`` references any class outside 

125 the allow-list (tampering or an unexpected format change). 

126 """ 

127 path = Path(folder_path) 

128 

129 # The .faiss file is a binary FAISS index (not pickle); read it as usual. 

130 faiss = dependable_faiss_import() 

131 index = faiss.read_index(str(path / f"{index_name}.faiss")) 

132 

133 # The .pkl is the only attack surface — unpickle it under restriction. 

134 with open(path / f"{index_name}.pkl", "rb") as f: 

135 docstore, index_to_docstore_id = _RestrictedFaissUnpickler(f).load() 

136 

137 return FAISS( 

138 embeddings, 

139 index, 

140 docstore, 

141 index_to_docstore_id, 

142 normalize_L2=normalize_L2, 

143 )
