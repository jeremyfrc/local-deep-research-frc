469 @staticmethod 

470 def _compute_version_hash(

472 content: Optional[str], 

473 tags: Optional[List[str]], 

474 ) -> str: 

475 """Compute the dedup hash for a NoteVersion snapshot. 

476 

477 JSON-encodes the whole ``[title, content, sorted-tags]`` triple and 

478 hashes that, so title-only / content-only / tag-only edits each get a 

479 distinct hash and aren't silently deduped against an existing snapshot. 

480 Tags are sorted first so reordering alone doesn't churn the hash. 

481

483 content can each contain a literal NUL byte — a JSON ``\\u0000`` in the 

484 request body decodes to ``\\x00`` and survives SQLite TEXT storage 

485 untouched (SQLite is length-prefixed, not NUL-terminated), and nothing 

486 validates it away — so a NUL separator is NOT injection-proof. Pre-fix, 

487 ``title="MyTitle", content="Hello\\x00World"`` and 

488 ``title="MyTitle\\x00Hello", content="World"`` produced the identical 

489 NUL-joined payload and hash, silently dedup-absorbing a genuinely 

490 different save out of version history. JSON escapes every control 

491 character (NUL included) as ``\\uXXXX`` and quotes/escapes each element, 

492 making the encoding injective across all three fields. 

493 

494 Hash-format note: this serialization differs from older builds', so the 

495 hash changes for previously-written snapshots. The only consumer is the 

496 (document_id, content_hash) dedup check, so the sole effect is a 

497 one-time extra version row the first time a note is re-saved in a state 

498 that already has an old-format snapshot — benign.

501 content_s = "" if content is None else content 

502 sorted_tags = sorted(t for t in (tags or []) if isinstance(t, str))

505 )

2884 def _resolve_link_internal( 

2885 self, session: Session, link_text: str 

2886 ) -> Optional[Dict[str, Any]]: 

2887 """Resolve a link to a note document. 

2888 

2889 Supports the Obsidian/Roam pipe alias syntax 

2890 ``[[Title|Display Text]]``: the target is everything before the 

2891 first ``|``; ``Display Text`` is rendering-only and ignored at 

2892 resolution time. Without this split, a link like 

2893 ``[[My Note|see here]]`` would search for a literal note titled 

2894 "my note|see here" — which never exists — and the link would 

2895 silently fall back to a stale cached target id on every save. 

2896 

2897 Two notes can have the same title (e.g. duplicate "TODO" notes). 

2898 Tie-break on ``created_at`` ascending so the link resolves to the 

2899 OLDEST match. Resolving to the newest instead reads as more 

2900 recency-intuitive, but it silently re-points every existing 

2901 ``[[Title]]`` the moment a new note reuses that title (link drift by 

2902 spooky-action-at-a-distance); oldest is stable and deterministic 

2903 (without the order_by, ``.first()`` is at the DB's discretion and can 

2904 flip between calls). Real per-target disambiguation — capturing the 

2905 picked note's id at autocomplete time — is tracked as follow-up work. 

2906 """ 

2907 # Strip pipe-alias before lookup. Use partition rather than 

2908 # split so a target containing additional pipes (rare but 

2909 # legal in markdown) doesn't get further mangled. 

2910 target_text, sep, _display = link_text.partition("|") 

2911 if sep: 

2912 link_text = target_text 

2913 link_text = link_text.strip() 

2914 # Degenerate target ("[[ ]]" / "[[|alias]]"): an empty key must 

2915 # never reach the lookup strategies — a LIKE '%' prefix pattern 

2916 # would "resolve" to EVERY note (deterministically the user's 

2917 # oldest one). The frontend treats an empty target as a literal, 

2918 # not a link; mirror that here. 

2919 if not link_text: 

2920 return None 

2921 source_type_id = self._get_note_source_type_id(session) 

2922 

2923 # Strategy 1: Exact title match. Fold BOTH sides in SQL: folding 

2924 # the link text in Python instead diverges for non-ASCII, because 

2925 # SQLite's lower() folds ASCII only while str.lower() folds full 

2926 # Unicode — a byte-identical title like "Über Alles" could then 

2927 # never match its own link (lower('Über Alles') = 'Über alles', 

2928 # but Python sent 'über alles'). SQL-side folding keeps ASCII 

2929 # case-insensitivity and makes exact-typed non-ASCII titles 

2930 # resolve. (Case-insensitive matching of the non-ASCII letters 

2931 # themselves would need a custom collation on the connection.) 

2932 document = ( 

2933 session.query(Document) 

2934 .filter( 

2935 Document.source_type_id == source_type_id, 

2936 func.lower(Document.title) == func.lower(link_text), 

2937 ) 

2938 .order_by(Document.created_at.asc(), Document.id.asc()) 

2939 .first() 

2940 ) 

2941 if document: 

2942 return {"id": document.id, "title": document.title} 

2943 

2944 return self._resolve_link_prefix(session, link_text, source_type_id)
